% Inline parameter name — e.g. \param{frq}
\newcommand{\param}[1]{\texttt{\textbf{#1}}}

% Inline menu path — e.g. \menu{OSC > Waveform}
\newcommand{\menu}[1]{\texttt{#1}}

% OLED display mockup — simulates the green-on-black display
% Usage: \oled{line1}{line2}
\newcommand{\oled}[2]{%
  \begin{center}
  \fboxsep=6pt
  \colorbox{lxrdark}{%
    \parbox{6cm}{%
      \ttfamily\small\color{lxrgreen}
      \hspace*{2pt}#1\\[2pt]
      \hspace*{2pt}#2
    }%
  }
  \end{center}%
}

% Single-line OLED display
\newcommand{\oledsingle}[1]{%
  \begin{center}
  \colorbox{lxrdark}{%
    \parbox{6cm}{%
      \ttfamily\small\color{lxrgreen}
      \hspace*{2pt}#1
    }%
  }
  \end{center}%
}

% Attention box (replaces the warning flask icon)
% Usage: \attention{text}
\newmdenv[
  backgroundcolor=attentionbg,
  linecolor=lxrorange,
  linewidth=2pt,
  leftline=true, rightline=false, topline=false, bottomline=false,
  innerleftmargin=10pt, innerrightmargin=8pt,
  innertopmargin=6pt, innerbottommargin=6pt,
  skipabove=8pt, skipbelow=8pt
]{attentionframe}

\newcommand{\attention}[1]{%
  \begin{attentionframe}
  \textbf{\textcolor{lxrorange}{Attention!}}\quad #1
  \end{attentionframe}%
}

% Tip / "Did you know?" box
% Usage: \tip{text}
\newmdenv[
  backgroundcolor=tipbg,
  linecolor=lxrblue,
  linewidth=2pt,
  leftline=true, rightline=false, topline=false, bottomline=false,
  innerleftmargin=10pt, innerrightmargin=8pt,
  innertopmargin=6pt, innerbottommargin=6pt,
  skipabove=8pt, skipbelow=8pt
]{tipframe}

\newcommand{\tip}[1]{%
  \begin{tipframe}
  \textbf{Did you know?}\quad #1
  \end{tipframe}%
}

% Parameter table environment — 3-col: Displayed name | Full name | Description
% Usage: wrap rows in \begin{paramtable}...\end{paramtable}
\newenvironment{paramtable}{%
  \begin{center}
  \small
  \begin{tabular}{>{\ttfamily}l  l  p{8cm}}
  \toprule
  \normalfont\textbf{Display} & \textbf{Name} & \textbf{Description} \\
  \midrule
}{%
  \bottomrule
  \end{tabular}
  \end{center}
}
